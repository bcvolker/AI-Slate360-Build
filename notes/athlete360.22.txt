lib/useAthlete360Store.ts (State: Roster, Sessions, Threads)
TypeScript'use client';
import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import { supabase } from '../lib/supabase';

type Position = 'QB' | 'WR' | 'OL' | 'DL' | 'LB' | 'DB' | 'RB' | 'K' | 'P';

interface Player { id: string; number: number; name: string; position: Position; }

interface Clip { id: string; playerId: string; type: 'rep' | 'highlight' | 'injury_risk'; status: 'processing' | 'ready'; notes?: string; threads?: any[]; } // 2-way

interface Session { id: string; name: string; date: string; clips: Clip[]; }

type Mode = 'quick' | 'full';

interface Athlete360State {
  roster: Player[];
  sessions: Session[];
  activeSessionId: string | null;
  mode: Mode;
  bufferStatus: 'connected' | 'disconnected';
  loading: boolean;

  // Actions
  setMode: (mode: Mode) => void;
  hydrateRoster: () => Promise<void>;
  startSession: (name: string) => Promise<void>;
  tagRep: (playerId: string, type: Clip['type']) => Promise<void>;
  processClip: (clipId: string, mode: Mode) => Promise<void>;
  addThread: (clipId: string, note: string, selfie?: string) => Promise<void>; // 2-way
  shareClip: (clipId: string) => Promise<string>;
  trackParticipation: (clipId: string, watchTime: number) => Promise<void>; // Homework
}

export const useAthlete360Store = create<Athlete360State>()(
  persist(
    (set, get) => ({
      roster: [
        { id: '1', number: 12, name: 'QB Smith', position: 'QB' },
        { id: '2', number: 1, name: 'WR Johnson', position: 'WR' },
        // OL/DL etc.
      ],
      sessions: [],
      activeSessionId: null,
      mode: 'quick',
      bufferStatus: 'disconnected',
      loading: false,

      setMode: (mode) => set({ mode }),

      hydrateRoster: async () => {
        const { data } = await supabase.from('athlete_players').select('*');
        set({ roster: data || [] });
      },

      startSession: async (name) => {
        set({ loading: true });
        const { data } = await supabase.from('athlete_sessions').insert({ name }).select().single();
        set((state) => ({ sessions: [...state.sessions, data], activeSessionId: data.id, bufferStatus: 'connected' }));
        await fetch('/api/athlete360/start-buffer', { method: 'POST', body: JSON.stringify({ sessionId: data.id }) });
        set({ loading: false });
      },

      tagRep: async (playerId, type) => {
        const { activeSessionId } = get();
        if (!activeSessionId) return;
        const { data } = await supabase.from('athlete_clips').insert({ session_id: activeSessionId, player_id: playerId, type }).select().single();
        set((state) => ({
          sessions: state.sessions.map(s => s.id === activeSessionId ? { ...s, clips: [{ ...data, status: 'processing', threads: [] }, ...s.clips] } : s),
        }));
      },

      processClip: async (clipId, mode) => {
        await fetch('/api/athlete360/process', { method: 'POST', body: JSON.stringify({ clipId, mode }) });
      },

      addThread: async (clipId, note, selfie) => {
        const { data } = await supabase.from('athlete_threads').insert({ clip_id: clipId, note, selfie_url: selfie }).select().single();
        set((state) => ({
          sessions: state.sessions.map(s => ({
            ...s,
            clips: s.clips.map(c => c.id === clipId ? { ...c, threads: [...c.threads || [], data] } : c)
          }))
        }));
      },

      shareClip: async (clipId) => {
        const { data } = await supabase.from('athlete_shares').insert({ clip_id: clipId }).select('url').single();
        return data.url;
      },

      trackParticipation: async (clipId, watchTime) => {
        await supabase.from('participation_logs').insert({ clip_id: clipId, watch_time: watchTime });
        if (watchTime < 80) {/* Nudge notification */ }
      },
    }),
    { name: 'athlete360-storage' }
  )
);
